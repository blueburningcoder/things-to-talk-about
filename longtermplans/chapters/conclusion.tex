\section{Zusammenfassung}

\subsection{Angesetzte Ziele}

\begin{frame}[c]{Ziele des Workshops}
    \Large
    \onslide<2-> \cmb Jeder schreibt seine Ziele auf \\
    \onslide<3-> \cmb Jeder macht sich sehr genaue Gedanken über seine Ziele \\
    \onslide<4-> \cmb Jeder sucht sich eine Vertrauensperson, um den Fortschritt regelmäßig zu teilen \\
\end{frame}


\begin{frame}[c]{Ablauf}
    \large
    \onslide<2-> \cmb Detailliertes Inventar der momentanen Situation \\
    \onslide<3-> \cmb Momentane Situation ist organisiert (Projekte, Kontexte) \\
    \onslide<4-> \cmb (Jahresrückblick) \\
    \onslide<5-> \cmb Neue Ziele sind gesetzt \\
    \onslide<6-> \cmb Ziele sind ausführlich geplant \\
    \onslide<7-> \cmb Verantwortungspartner ist gefunden \\
\end{frame}
